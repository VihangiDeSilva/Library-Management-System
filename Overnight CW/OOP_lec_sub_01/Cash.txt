package myapplication;

import java.util.ArrayList;
import java.util.List;

public class CashRegister  {

   private List <RetailItem> retailItemsList =new ArrayList<>();

   public double getTotal(){

       double total=0;

       for(RetailItem item:retailItemsList){
           total+=item.getPrice();
       }
       return total;
   }
   public void clear(){
       retailItemsList.clear();
   }

   public void purchaseIteam(RetailItem item){
       retailItemsList.add(item);
   }
}
